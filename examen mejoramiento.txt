SELECT 
    egresados.id,
    egresados.nombre,
    egresados.telefono,
    egresados.correo,
    empresa_egresados.Es_Actual,
    empresa_egresados.Area_empresa
FROM egresados
JOIN empresa_egresados ON egresados.id = empresa_egresados.id;

SELECT
courses.id AS course_id,
courses.titulo,
courses.descripcion,
courses.precio,
categories.id AS category_id,
categories.nombre
FROM courses
INNER JOIN categories 
ON courses.category_id = categories.id;

SELECT * 
FROM courses
WHERE id = (
    SELECT id 
    FROM courses
    WHERE id = 4
);
